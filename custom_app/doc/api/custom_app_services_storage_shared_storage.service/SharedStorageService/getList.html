Future getList({@required String key}) async {
  SharedPreferences shared = await SharedPreferences.getInstance();
  var data = shared.get(key);

  if (data != null) {
    Iterable decoded = json.decode(data);
    var result = decoded;
    return result;
  } else {
    return null;
  }
}
